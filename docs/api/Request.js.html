import qs from 'qs';

/**
 * Build request parameters and allow them to be checked against a string (usually the current path)
 *
 * @param {string} Route
 * @param {array} keys
 * @param {RegExp} regex
 */
class Request {

    constructor(route, keys, regex) {
        this.route = route;
        this.keys = keys;
        this.regex = regex;
    }

    /**
     * Match a path string -- returns a request object if there is a match -- returns false otherwise
     *
     * @return {Object} req
     */
    parse(path) {

        var parts = path.split('?');
        var query;
        var str;
        var match;
        var req;

        str = parts.shift();
        query = parts.join('');
        match = str.match(this.regex);

        req = {
            params: {},
            keys: this.keys,
            matches: (match || []).slice(1),
            match: match,
            query: qs.parse(query)
        };
        req.matches.forEach((value, i) => {

            var key = (this.keys[i] && this.keys[i].name) ? this.keys[i].name : i;
            // Parameter key will be its key or the iteration index. This is useful if a wildcard (*) is matched
            req.params[key] = (value) ? decodeURIComponent(value) : undefined;
        });

        return req;
    }
}

export default Request
