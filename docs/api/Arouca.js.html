 * @property {Array<Route>} routes

/**
 * @typedef {Object} Route
 * @property {string} href The uri this route will match.
 * @property {*} view A value representing the view (handled by framework).
 * @property {string} controller The name of the controller for this route.
 */

/**
 * Arouca is the main class of this library.
 */
class Arouca {

    constructor(router) {

        this.controllers = {};
        this.routes = [];
        this.router = router;
        this.routingLogic = () => {
            throw new Error('No routing logic provided!');
        }

    }

    /**
     * controller adds a controller to the internal list.
     * @param {string} name
     * @param {function} controller
     * @returns {Arouca}
     */
    controller(name, value) {

        this.controllers[name] = value;
        return this;

    }

    /**
     * route adds a route to the internal stack.
     * @param {RoutingDef} def
     * @returns {Arouca}
     */
    route(def) {

        if (!Array.isArray(def.routes))
            throw new Error('Routing definition without routes section received!');

        def.routes.forEach(r => {
            this.routes.push(r);
        });


    }

    /**
     * use sets the logic for executing routes.
     * It is intentionally left to the calling code 
     * to provide the route executing logic in order
     * to prevent this library from being tied down
     * to a particular framework or library.
     *
     * The logic function should not be bound
     * as it is called in the context of this class.
     * @param {function} logic
     * @returns {Arouca}
     */
    use(logic) {

        this.routingLogic = logic;
        return this;

    }

    /**
     * start the routing engine.
     */
    start() {
        this.routes.forEach(this.routingLogic);
    }

}

export default Arouca
